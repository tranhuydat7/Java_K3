create database DB_Fastfood_System;
use  DB_Fastfood_System;

create table Customers(
Customer_id int auto_increment primary key,
name varchar(255),
email varchar(255),
phone varchar(11),
address varchar(255),
birthday date,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table categories(
category_id int auto_increment primary key,
name varchar(255),
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);



create table cities(
city_id int auto_increment primary key,
name varchar(255),
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table districts(
district_id int auto_increment primary key,
name varchar(255),
city_id int,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table store_products(
store_product_id int auto_increment primary key,
store_id int,
product_id int,
status int,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table roles(
roles_id int auto_increment primary key,
name varchar(255),
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table users(
user_id int auto_increment primary key,
code int(6),
name varchar(255),
password varchar(255),
email varchar(255),
phone varchar(11),
avatar varchar(255),
role_id int,
store_id int,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table stores(
store_id int auto_increment primary key,
name varchar(255),
address varchar(255),
district_id int,
img varchar(255),
description text,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table products(
product_id int auto_increment primary key,
name varchar(255),
description text,
category_id int,
image varchar(255),
price double,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table shipping_units(
shipping_unit_id int auto_increment primary key,
name varchar(255),
address varchar(255),
phone varchar(11),
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table promotions(
promotion_id int auto_increment primary key,
name varchar(255),
start_date datetime,
end_date datetime,
apply_day varchar(255),
day_id int,
discount int,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table orders(
order_id int auto_increment primary key,
store_id int,
customer_id int,
promotion_id int,
shipping_unit_id int,
price double,
payment_method int,
status int,
payment_status int,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table order_details(
order_detail_id int auto_increment primary key,
order_id int,
store_product_id int,
quantity int,
created_at timestamp default now(),
updated_at timestamp default now() on update now()
);
create table Days(
day_id int auto_increment primary key,
name varchar(6));
